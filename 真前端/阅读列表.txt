<script>
document.addEventListener('DOMContentLoaded', () => {
    const recommendList = document.querySelector('.recommend-list');

    // ✅ 改用新的分页接口
    const API_BASE_URL = 'https://api.anchorx.ca/api/works/public';

    // 分页 & 状态
    let currentPage = 1;
    const PAGE_SIZE = 5;   // 每页 5 条
    let isLoading = false;
    let hasMore = true;

    // 进度条 DOM
    const progressContainer = document.getElementById('load-progress-container');
    const progressText = document.getElementById('load-progress-text');
    const progressBar = document.getElementById('load-progress-bar');

    /**
     * 获取作品的更新时间（这里直接用 updatedAt）
     */
    function getLatestUpdateTime(work) {
        return work.updatedAt || work.createdAt || null;
    }

    /**
     * 根据作品对象生成推荐条目的 HTML 片段
     */
    function createWorkItemHtml(work) {
        const authorName = work.author ? work.author.username : '✍️用户';
        const latestDateString = getLatestUpdateTime(work);
        const date = latestDateString ? new Date(latestDateString).toLocaleDateString() : '未知';

        // 直接用后端的字数字段（如果有）
        const wordCount = typeof work.wordCount === 'number' ? work.wordCount : 0;

        // 从第一页内容截取一段摘要
        let snippet = '';
        if (Array.isArray(work.content) && work.content.length > 0) {
            const firstPage = work.content[0];
            if (firstPage && firstPage.content) {
                let text = '';
                if (typeof firstPage.content === 'string') {
                    text = firstPage.content;
                } else if (typeof firstPage.content === 'object' && Array.isArray(firstPage.content.ops)) {
                    text = firstPage.content.ops
                        .map(op => typeof op.insert === 'string' ? op.insert : '')
                        .join('');
                }
                const cleanSnippet = text
                    .replace(/[\n\r\t\u200B-\u200D\uFEFF#*`]/g, '')
                    .trim();
                snippet = cleanSnippet.substring(0, 100) + (cleanSnippet.length > 100 ? '...' : '');
            }
        } else {
            snippet = '已发布安科小说的摘要内容...';
        }

        const coverImageSrc = work.coverImage
            ? work.coverImage
            : "https://i.ibb.co/QFPHMF8T/Copilot-20250922-174349.png";

        const statusTagHtml = work.isPublished
            ? '<span class="tag tag-published">✅ 已发布</span>'
            : '<span class="tag tag-draft">✏️ 草稿</span>';

        return `
            <article class="recommend-item" data-id="${work._id}">
                <a href="https://zhidianworld.com/read/?id=${work._id}" class="work-title-link">
                    <h4 class="recommend-title">${work.title}</h4>
                </a>
                <p class="recommend-info">
                    <span class="info-author">✍️ ${authorName}</span>
                    <span class="info-update">⏰ ${date}更新</span>
                    <span class="info-stats">
                        <span class="info-views"><span class="icon">&#x1F441;</span>${work.views || 0}</span>
                        <span class="info-likes"><span class="icon">&#x2764;</span>${work.likesCount || 0}</span>
                    </span>
                </p>
                <div class="recommend-content">
                    <p class="content-text">${snippet}</p>
                    <img src="${coverImageSrc}" alt="安科封面" class="content-image">
                </div>
                <div class="recommend-bottom">
                    <span class="word-count">安科·${wordCount}字</span>
                    <div class="ankou-tags">
                        <span class="tag">安科</span>
                        ${statusTagHtml}
                    </div>
                </div>
            </article>
        `;
    }

    /**
     * 加载下一页作品（初次加载 / 懒加载共用）
     * @param {boolean} initial 是否首页第一次加载
     */
    async function loadNextPage(initial = false) {
        if (isLoading || !hasMore) return;
        isLoading = true;

        try {
            if (initial && progressContainer) {
                progressContainer.style.display = 'block';
                progressText.textContent = '正在请求数据...';
                progressBar.style.width = '5%';
            }

            const url = `${API_BASE_URL}?page=${currentPage}&limit=${PAGE_SIZE}`;
            const response = await fetch(url, { method: 'GET' });
            if (!response.ok) throw new Error('网络请求失败');

            if (initial && progressBar) progressBar.style.width = '20%';

            const data = await response.json();
            const works = Array.isArray(data.works) ? data.works : [];

            if (initial) {
                recommendList.innerHTML = '';
            }

            if (works.length === 0) {
                if (initial) {
                    recommendList.innerHTML = '<p style="text-align:center;color:#999;margin-top:50px;">暂无作品。</p>';
                }
                hasMore = false;
                return;
            }

            // 追加渲染本页的作品（5 条）
            works.forEach(work => {
                const html = createWorkItemHtml(work);
                recommendList.insertAdjacentHTML('beforeend', html);
            });

            currentPage += 1;
            hasMore = !!data.hasMore;

            if (initial && progressBar) {
                progressText.textContent = '加载完成';
                progressBar.style.width = '100%';
                setTimeout(() => {
                    progressContainer.style.display = 'none';
                    progressBar.style.width = '0%';
                }, 500);
            }
        } catch (error) {
            console.error('加载作品失败:', error);
            if (initial) {
                recommendList.innerHTML =
                    '<p style="text-align:center;color:red;margin-top:50px;">加载失败，请刷新页面。</p>';
                if (progressContainer) progressContainer.style.display = 'none';
            }
        } finally {
            isLoading = false;
        }
    }

    // 监听滚动：快到底部时再加载下一页
    window.addEventListener('scroll', () => {
        if (!hasMore || isLoading) return;

        const scrollBottom = window.innerHeight + window.scrollY;
        const docHeight = document.body.offsetHeight;

        // 离底部 200px 以内就触发懒加载
        if (scrollBottom >= docHeight - 200) {
            loadNextPage(false);
        }
    });

    // 首次加载：只拉第一页（5 条）
    loadNextPage(true);
});
</script>
