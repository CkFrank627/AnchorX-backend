  const API_BASE = 'https://api.anchorx.ca';  // 统一放前面

const DEFAULT_AVATAR_URL = 'https://i.ibb.co/GQWbDBGv/black-cat-icon-0.png';  // 你图床的默认头像（建议换成直链）
  const IMGBB_API_KEY = '040dbc9e6e680f321e09d9e05f447094';
  
// ==================== TAB 切换 ====================
document.querySelectorAll(".tab-btn").forEach(btn => {
  btn.addEventListener("click", () => {
    document.querySelectorAll(".tab-btn").forEach(b => b.classList.remove("active"));
    document.querySelectorAll(".tab-content").forEach(c => c.classList.remove("active"));

    if (btn.dataset.tab === "message") loadMessages();
  });

// ==================== 修改用户名 ====================
document.getElementById("change-username-btn").onclick = async () => {
  const newName = document.getElementById("new-username").value.trim();
  const token = localStorage.getItem("token");

  if (!newName) return alert("请输入新用户名！");
  if (!token) return alert("请先登录");

  const res = await fetch(`${API_BASE}/api/users/profile`, {
    method: "PATCH",
    headers: {
      "Content-Type": "application/json",
      "Authorization": `Bearer ${token}`
    },
    body: JSON.stringify({ username: newName })
  });

  const data = await res.json();
  alert(data.message || "修改成功，请重新登录");
};

// ==================== 更改密码 ====================
document.getElementById("change-password-btn").onclick = async () => {
  const oldPassword = document.getElementById("old-password").value.trim();
  const newPassword = document.getElementById("new-password").value.trim();
  const confirmPassword = document.getElementById("confirm-new-password").value.trim();
  const token = localStorage.getItem("token");

  if (!oldPassword || !newPassword || !confirmPassword)
    return alert("请填写所有字段");

  if (newPassword !== confirmPassword)
    return alert("两次密码不一致");

  const res = await fetch(`${API_BASE}/api/users/change-password`, {
    method: "PATCH",
    headers: {
      "Content-Type": "application/json",
      "Authorization": `Bearer ${token}`
    },
    body: JSON.stringify({ oldPassword, newPassword })
  });

  const data = await res.json();
  alert(data.message);
};

// ==================== 消息箱 ====================
async function loadMessages() {
  const box = document.getElementById("message-box");
  const token = localStorage.getItem("token");

  if (!token) {
    box.innerHTML = "<p>请登录查看消息</p>";
    return;
  }

  box.innerHTML = "<p>加载中...</p>";

  const res = await fetch("https://api.anchorx.ca/api/notifications", {
    headers: { "Authorization": `Bearer ${token}` }
  });

  const data = await res.json();

  if (!Array.isArray(data) || data.length === 0) {
    box.innerHTML = "<p>暂无消息</p>";
    return;
  }

  box.innerHTML = "";

  data.forEach(n => {
    const item = document.createElement("div");
    item.className = "message-item";
    item.innerHTML = `
      <b>${n.sender?.username || "未知用户"}</b>：${n.message}<br>
      <small>${new Date(n.createdAt).toLocaleString()}</small>
    `;
    box.appendChild(item);
  });
}

// ==================== 头像上传 / 裁剪 ====================
let avatarCropper = null;
const avatarPreviewImg = document.getElementById('avatarPreviewImg');
const avatarFileInput = document.getElementById('avatarFileInput');
const avatarCropModal = document.getElementById('avatarCropModal');
const avatarCropImage = document.getElementById('avatarCropImage');
const avatarCropCancel = document.getElementById('avatarCropCancel');
const avatarCropConfirm = document.getElementById('avatarCropConfirm');

function getToken() {
  return localStorage.getItem('token');
}

(async function loadProfileForAvatar() {
  const token = getToken();
  if (!token) return;
  try {
    const res = await fetch(`${API_BASE}/api/users/profile`, {
      headers: { 'Authorization': `Bearer ${token}` }
    });

    if (!res.ok) {
      console.warn('加载头像失败，状态码 =', res.status);
      avatarPreviewImg.src = DEFAULT_AVATAR_URL;
      return;
    }

    const data = await res.json();

    if (data.user && data.user.avatarUrl) {
      const baseUrl = data.user.avatarUrl;
      // 加个时间戳避免缓存，imgbb 也没问题
      const newUrl = `${baseUrl}${baseUrl.includes('?') ? '&' : '?'}t=${Date.now()}`;
      avatarPreviewImg.src = newUrl;
    } else {
      avatarPreviewImg.src = DEFAULT_AVATAR_URL;
    }

  } catch (e) {
    console.warn('加载头像失败:', e);
    avatarPreviewImg.src = DEFAULT_AVATAR_URL;
  }
})();



// 选择文件后，弹出裁剪框
avatarFileInput.addEventListener('change', (e) => {
  const file = e.target.files && e.target.files[0];
  if (!file) return;

  const reader = new FileReader();
  reader.onload = function(evt) {
    avatarCropImage.src = evt.target.result;
    avatarCropModal.style.display = 'flex';

    // 销毁旧 cropper
    if (avatarCropper) {
      avatarCropper.destroy();
    }
    // 初始化 cropper（固定 1:1）
    avatarCropper = new Cropper(avatarCropImage, {
      aspectRatio: 1,
      viewMode: 1,
      dragMode: 'move',
      autoCropArea: 1,
    });
  };
  reader.readAsDataURL(file);
});

// 取消裁剪
avatarCropCancel.addEventListener('click', () => {
  avatarCropModal.style.display = 'none';
  if (avatarCropper) {
    avatarCropper.destroy();
    avatarCropper = null;
  }
});

// 确认裁剪并上传（前端直传 imgbb，再把链接保存给后端）
avatarCropConfirm.addEventListener('click', async () => {
  const token = getToken();
  if (!token) {
    alert('请先登录');
    return;
  }
  if (!avatarCropper) return;

  try {
    const canvas = avatarCropper.getCroppedCanvas({
      width: 256,
      height: 256,
    });

    // 1️⃣ 先用本地裁剪结果更新预览，不用等任何服务器
    const dataUrl = canvas.toDataURL('image/png'); // "data:image/png;base64,..."
    avatarPreviewImg.src = dataUrl;

    // 2️⃣ 提取 base64 数据部分
    const base64Data = dataUrl.split(',')[1]; // 去掉前缀 "data:image/png;base64,"

    // 3️⃣ 调用 imgbb API（前端直传）
    if (!IMGBB_API_KEY || IMGBB_API_KEY === '在 imgBB 后台复制来的完整 API Key') {
      alert('尚未配置 imgbb API key');
      return;
    }

    const formData = new FormData();
    formData.append('image', base64Data);

    const imgbbRes = await fetch(`https://api.imgbb.com/1/upload?key=${IMGBB_API_KEY}`, {
      method: 'POST',
      body: formData
    });

    const imgbbData = await imgbbRes.json();
    if (!imgbbRes.ok || !imgbbData.success) {
      console.error('imgbb 上传失败:', imgbbData);
      alert('上传到图床失败，请稍后重试');
      return;
    }

    const imageUrl = imgbbData.data.display_url || imgbbData.data.url;
    if (!imageUrl) {
      alert('图床未返回有效图片链接');
      return;
    }

    // 4️⃣ 把图床链接保存到后端用户资料（只存 URL，不上传文件）
    const saveRes = await fetch(`${API_BASE}/api/users/profile`, {
      method: 'PATCH',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${token}`
      },
      body: JSON.stringify({ avatarUrl: imageUrl })
    });

    const saveData = await saveRes.json();
    if (!saveRes.ok) {
      console.error('保存头像链接失败:', saveData);
      alert(saveData.message || '保存头像链接失败');
      return;
    }

    // 5️⃣ 再用一次图床 URL + 时间戳 刷新预览，避免缓存问题
    const finalUrl = `${imageUrl}${imageUrl.includes('?') ? '&' : '?'}t=${Date.now()}`;
    avatarPreviewImg.src = finalUrl;

    alert('头像更新成功！');

  } catch (e) {
    console.error('上传头像出错:', e);
    alert('头像上传失败：' + (e.message || '网络错误'));
  } finally {
    avatarCropModal.style.display = 'none';
    if (avatarCropper) {
      avatarCropper.destroy();
      avatarCropper = null;
    }
  }
});

  avatarFileInput.click();
});
