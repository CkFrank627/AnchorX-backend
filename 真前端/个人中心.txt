<script>
// ==================== TAB 切换 ====================
document.querySelectorAll(".tab-btn").forEach(btn => {
  btn.addEventListener("click", () => {
    document.querySelectorAll(".tab-btn").forEach(b => b.classList.remove("active"));
    document.querySelectorAll(".tab-content").forEach(c => c.classList.remove("active"));

    btn.classList.add("active");
    document.getElementById("tab-" + btn.dataset.tab).classList.add("active");

    if (btn.dataset.tab === "message") loadMessages();
  });
});

// ==================== 修改用户名 ====================
document.getElementById("change-username-btn").onclick = async () => {
  const newName = document.getElementById("new-username").value.trim();
  const token = localStorage.getItem("token");

  if (!newName) return alert("请输入新用户名！");
  if (!token) return alert("请先登录");

  const res = await fetch("/api/users/profile", {
    method: "PATCH",
    headers: {
      "Content-Type": "application/json",
      "Authorization": `Bearer ${token}`
    },
    body: JSON.stringify({ username: newName })
  });

  const data = await res.json();
  alert(data.message || "修改成功，请重新登录");
};

// ==================== 更改密码 ====================
document.getElementById("change-password-btn").onclick = async () => {
  const oldPassword = document.getElementById("old-password").value.trim();
  const newPassword = document.getElementById("new-password").value.trim();
  const confirmPassword = document.getElementById("confirm-new-password").value.trim();
  const token = localStorage.getItem("token");

  if (!oldPassword || !newPassword || !confirmPassword)
    return alert("请填写所有字段");

  if (newPassword !== confirmPassword)
    return alert("两次密码不一致");

  const res = await fetch("/api/users/change-password", {
    method: "PATCH",
    headers: {
      "Content-Type": "application/json",
      "Authorization": `Bearer ${token}`
    },
    body: JSON.stringify({ oldPassword, newPassword })
  });

  const data = await res.json();
  alert(data.message);
};

// ==================== 消息箱 ====================
async function loadMessages() {
  const box = document.getElementById("message-box");
  const token = localStorage.getItem("token");

  if (!token) {
    box.innerHTML = "<p>请登录查看消息</p>";
    return;
  }

  box.innerHTML = "<p>加载中...</p>";

  const res = await fetch("https://api.anchorx.ca/api/notifications", {
    headers: { "Authorization": `Bearer ${token}` }
  });

  const data = await res.json();

  if (!Array.isArray(data) || data.length === 0) {
    box.innerHTML = "<p>暂无消息</p>";
    return;
  }

  box.innerHTML = "";

  data.forEach(n => {
    const item = document.createElement("div");
    item.className = "message-item";
    item.innerHTML = `
      <b>${n.sender?.username || "未知用户"}</b>：${n.message}<br>
      <small>${new Date(n.createdAt).toLocaleString()}</small>
    `;
    box.appendChild(item);
  });
}
</script>
