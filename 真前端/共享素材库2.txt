async function handleDownload(event) {
    event.stopPropagation();
    const galleryId = event.target.dataset.id;
    const galleryTitle = event.target.dataset.title || 'images';
    const button = event.target.closest('.download-btn');
    const originalContent = button.innerHTML;

    button.disabled = true;
    button.innerHTML = '<span class="loading-spinner"></span>';

    // ğŸŸ¦ è·å–è¿›åº¦æ¡ DOM
    const progressContainer = document.getElementById('download-progress-container');
    const progressText = document.getElementById('download-progress-text');
    const progressBar = document.getElementById('download-progress-bar');
    progressContainer.style.display = 'block';
    progressText.textContent = 'æ­£åœ¨å‡†å¤‡ä¸‹è½½...';
    progressBar.style.width = '0%';

    try {
        const response = await fetch(`${API_BASE_URL}/api/galleries/${galleryId}`);
        if (!response.ok) throw new Error('è·å–å›¾åº“æ•°æ®å¤±è´¥ã€‚');
        const gallery = await response.json();

        if (!gallery.images || gallery.images.length === 0) {
            alert('è¯¥å›¾åº“æ²¡æœ‰å›¾ç‰‡å¯ä»¥ä¸‹è½½ã€‚');
            progressContainer.style.display = 'none';
            return;
        }

        const zip = new JSZip();
        const totalImages = gallery.images.length;
        let downloadedCount = 0;

        for (const image of gallery.images) {
            const imageUrl = image.url;
            const imageResponse = await fetch(imageUrl);
            const blob = await imageResponse.blob();
            const filename = imageUrl.substring(imageUrl.lastIndexOf('/') + 1);
            zip.file(filename, blob, { binary: true });

            downloadedCount++;
            const percent = Math.round((downloadedCount / totalImages) * 100);
            progressText.textContent = `æ­£åœ¨ä¸‹è½½å›¾ç‰‡ï¼š${downloadedCount} / ${totalImages} (${percent}%)`;
            progressBar.style.width = `${percent}%`;
        }

        progressText.textContent = 'æ­£åœ¨æ‰“åŒ… ZIP æ–‡ä»¶...';
        progressBar.style.width = '100%';

        const content = await zip.generateAsync({ type: 'blob' });
        saveAs(content, `${galleryTitle}.zip`);

        progressText.textContent = 'ä¸‹è½½å·²å®Œæˆï¼';
        progressBar.style.background = '#4caf50';
        setTimeout(() => {
            progressContainer.style.display = 'none';
            progressBar.style.background = '#0077cc';
        }, 1500);

    } catch (error) {
        console.error('ä¸‹è½½å›¾åº“å¤±è´¥:', error);
        alert('ä¸‹è½½å¤±è´¥ï¼Œè¯·ç¨åé‡è¯•ã€‚');
        progressContainer.style.display = 'none';
    } finally {
        button.disabled = false;
        button.innerHTML = originalContent;
    }
}
